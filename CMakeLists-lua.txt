SET( LUA_FOUND true )
IF (WIN32)
  SET( LUA_INCLUDE_DIR ../../os_win/include/MechatronixCore )
  IF ( BUILD_SHARED )
    find_library( LUA_LIBRARIES lua_win_${BITS}${DEBUG_SUFFIX}.lib HINTS ../../os_win/lib )
  ENDIF()
ENDIF (WIN32)
IF (UNIX)
  SET( LUA_INCLUDE_DIR ../../os_linux/include/MechatronixCore )
  IF ( BUILD_SHARED )
    find_library( LUA_LIBRARIES libLua_linux${DEBUG_SUFFIX}.so HINTS ../../os_linux/dll )
  ENDIF()
ENDIF (UNIX)
IF (APPLE)
  SET( LUA_INCLUDE_DIR ../../os_mac/include/MechatronixCore )
  IF ( BUILD_SHARED )
    find_library( LUA_LIBRARIES libLua_osx${DEBUG_SUFFIX}.dylib HINTS ../../os_mac/dll )
  ENDIF()
ENDIF(APPLE)
